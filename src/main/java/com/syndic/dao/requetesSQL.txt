CREATE TABLE paiement (
    id_paiement INT AUTO_INCREMENT PRIMARY KEY,
    mois VARCHAR(20) NOT NULL,
    paye BOOLEAN NOT NULL,

    FOREIGN KEY (id_user) REFERENCES users(id)
);

    checkadmin TINYINT(1) NOT NULL,
